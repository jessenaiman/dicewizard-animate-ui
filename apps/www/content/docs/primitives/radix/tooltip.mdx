## Usage

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger>Open tooltip</TooltipTrigger>
    <TooltipPortal>
      <TooltipContent>
        <p>Tooltip content</p>
      </TooltipContent>
    </TooltipPortal>
  </Tooltip>
</TooltipProvider>
```
